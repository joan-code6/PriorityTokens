Phase 2 — Tokenizer extension + embed init
Add the 11 special tokens and initialize their embeddings. Pure CPU
work, ~30 minutes.
Step 1 of 3
Add special tokens to the Qwen3 tokenizer
add_special_tokens() adds them to vocab and updates the fast BPE
tokenizer. Verify round-trip encoding works.
from transformers import AutoTokenizer

tok = AutoTokenizer.from_pretrained("Qwen/Qwen3-8B")

new_tokens = [f"<<Priority{i}>>" for i in range(1, 11)] + ["<<PrioEnd>>"]
tok.add_special_tokens({"additional_special_tokens": new_tokens})

# Verify: these should each be single tokens now
for t in new_tokens:
    ids = tok.encode(t, add_special_tokens=False)
    assert len(ids) == 1, f"{t} is not a single token: {ids}"

tok.save_pretrained("tokenizer/qwen3-8b-prio")
Step 2 of 3
Resize model embeddings and initialize from nearby token
Copy embedding vector from <|im_start|> — a special token the model
already understands. Do NOT use random init.
from transformers import AutoModelForCausalLM
import torch

model = AutoModelForCausalLM.from_pretrained("Qwen/Qwen3-8B",
                                              torch_dtype=torch.float32)
model.resize_token_embeddings(len(tok))

# Find <|im_start|> id in original tokenizer
im_start_id = tok.convert_tokens_to_ids("<|im_start|>")
src_embed = model.model.embed_tokens.weight[im_start_id].clone()

# Initialize all 11 new tokens from that embedding
for token in new_tokens:
    new_id = tok.convert_tokens_to_ids(token)
    with torch.no_grad():
        model.model.embed_tokens.weight[new_id] = src_embed.clone()

model.save_pretrained("tokenizer/qwen3-8b-prio")

Phase 3 — Dataset generation
Generate ~1,500 training examples using Claude API (Haiku). Estimated
cost: €2–5.
Step 1 of 5
Write the document component generator
Call Haiku to generate filler paragraphs and fact pairs separately.
Assemble in Python — never ask Claude to generate the whole tagged
document at once.
# data/generate.py — component approach
import anthropic, random, json

client = anthropic.Anthropic()

def gen_filler(topic, n_paragraphs=4) -> str:
    r = client.messages.create(
        model="claude-haiku-4-5",
        max_tokens=600,
        messages=[{"role":"user","content":
            f"Write {n_paragraphs} paragraphs about {topic}. "
            "Use specific but fictional names, numbers, dates. "
            "Plain prose, no headers."}]
    )
    return r.content[0].text

def gen_fact_pair(topic) -> dict:
    r = client.messages.create(
        model="claude-haiku-4-5",
        max_tokens=200,
        messages=[{"role":"user","content":
            f"For topic: {topic}. Give me:
"
            "- fact_a: one sentence with a specific measurable value
"
            "- fact_b: same sentence but a DIFFERENT value (plausible)
"
            "- question: a question whose answer is the value
"
            "Respond as JSON only."}]
    )
    return json.loads(r.content[0].text)
Step 2 of 5
Write the document assembler
Place facts at randomized positions. Track positions in metadata.
Enforce minimum 500-token gap between P10 and P5 spans.
def assemble_contrastive_pair(topic):
    filler_chunks = [gen_filler(topic) for _ in range(5)]
    pair = gen_fact_pair(topic)

    # Randomize positions — never let P10 always be last
    positions = random.sample(range(5), 2)
    p10_pos, p5_pos = sorted(positions, key=lambda _: random.random())

    chunks = filler_chunks.copy()
    chunks[p10_pos] = f"<<Priority10>>{pair['fact_a']}<<PrioEnd>> " + chunks[p10_pos]
    chunks[p5_pos]  = f"<<Priority5>>{pair['fact_b']}<<PrioEnd>> " + chunks[p5_pos]
    doc = "

Step 2 of 5
Configure QLoRA training with Unsloth
Start conservative. LoRA rank=16, alpha=32. You can increase rank if
behavior doesn't emerge, but start low to protect budget.
# train/config.py
from unsloth import FastLanguageModel

model, tokenizer = FastLanguageModel.from_pretrained(
    model_name = "tokenizer/qwen3-8b-prio",  # your extended tokenizer
    max_seq_length = 8192,
    dtype = None,  # auto-detect
    load_in_4bit = True,
)

model = FastLanguageModel.get_peft_model(
    model,
    r = 16,           # LoRA rank — start here
    target_modules = ["q_proj","k_proj","v_proj","o_proj",
                       "gate_proj","up_proj","down_proj"],
    lora_alpha = 32,
    lora_dropout = 0.05,
    bias = "none",
    use_gradient_checkpointing = True,
)
# IMPORTANT: also train the new token embeddings
# Unsloth may freeze embeddings by default — verify they're unfrozen
model.model.embed_tokens.weight.requires_grad_(True)
